128  if (Rows == Dynamic && Cols == Dynamic) {

149  throw runtime_error(stream.str().c_str());

156  throw runtime_error(stream.str().c_str());

216  }

221  Eigen::MatrixXd sub(num_monomials, 2);

234  sub(index, 1) = j + 1;
