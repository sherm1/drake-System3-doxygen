395  const double& t,
396  /* todo: add force state here */ const Eigen::VectorXd& u,
397  const KinematicsCache<double>& rigid_body_state) const override {
398  using namespace Eigen;
399  const Vector3d origin = Vector3d::Zero();
400  Vector3d xA_in_B = sys.getRigidBodyTree()->transformPoints(
401  rigid_body_state, origin, frameA->frame_index, frameB->frame_index);
402  Vector3d xB_in_A = sys.getRigidBodyTree()->transformPoints(
403  rigid_body_state, origin, frameB->frame_index, frameA->frame_index);
404  auto JA_in_B = sys.getRigidBodyTree()->transformPointsJacobian(
405  rigid_body_state, origin, frameA->frame_index, frameB->frame_index,
406  false);
407 
408  double length = xA_in_B.norm();
409  double vel =
410  (JA_in_B * rigid_body_state.getV()).dot(xA_in_B) / (length + EPSILON);
411  double force_magnitude = stiffness * (length - rest_length) + damping * vel;
412 
413  // std::cout << "l=" << length << ", v=" << vel << ", f=" <<
414  // force_magnitude << std::endl;
415 
416  Matrix<double, 6, 1> force = Matrix<double, 6, 1>::Zero();
417 
418  // apply (force_magnitude/length)*xA_in_B to B
419  force.tail<3>() = (force_magnitude / (length + EPSILON)) * xA_in_B;
420  auto tau = spatialForceInFrameToJointTorque(
421  sys.getRigidBodyTree().get(), rigid_body_state, frameB.get(), force);
422 
423  // apply (force_magnitude/length)*xB_in_A to A
424  force.tail<3>() = (force_magnitude / (length + EPSILON)) * xB_in_A;
425  tau += spatialForceInFrameToJointTorque(
426  sys.getRigidBodyTree().get(), rigid_body_state, frameA.get(), force);
427  return tau;
428  }
