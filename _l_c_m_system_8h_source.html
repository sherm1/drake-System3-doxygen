227 void runLCM(std::shared_ptr<System> sys, std::shared_ptr<lcm::LCM> lcm,
228  double t0, double tf,
229  const typename System::template StateVector<double> &x0,
230  const SimulationOptions &options = default_simulation_options) {
231  if (!lcm->good()) throw std::runtime_error("bad LCM reference");
232 
233  // typename System::template OutputVector<double> x = 1; // useful for
234  // debugging
235  auto lcm_input =
236  std::make_shared<internal::LCMInputSystem<System::template InputVector> >(
237  *sys, lcm);
238  auto lcm_output = std::make_shared<
239  internal::LCMOutputSystem<System::template OutputVector> >(lcm);
240  auto lcm_sys = cascade(lcm_input, cascade(sys, lcm_output));
241 
242  bool has_lcm_input =
243  internal::LCMInputSystem<System::template InputVector>::has_lcm_input;
244 
245  if (has_lcm_input && size(x0) == 0 && !sys->isTimeVarying()) {
246  // then this is really a static function, not a dynamical system.
247  // block on receiving lcm input and process the output exactly when a new
248  // input message is received.
249  // note: this will never return (unless there is an lcm error)
250 
251  // std::cout << "LCM output will be triggered on receipt of an LCM
252  // Input" << std::endl;
253 
254  double t = 0.0;
255  typename System::template StateVector<double> x;
256  NullVector<double> u;
257 
258  while (1) {
259  if (lcm->handle() != 0) {
260  throw std::runtime_error("something went wrong in lcm.handle");
261  }
262  lcm_sys->output(t, x, u);
263  }
264  } else {
265  internal::LCMLoop lcm_loop(*lcm);
266  std::thread lcm_thread;
267  if (has_lcm_input) {
268  // only start up the listener thread if I actually have inputs to listen
269  // to
270  lcm_thread = std::thread(&internal::LCMLoop::loopWithSelect, &lcm_loop);
271  }
272 
273  SimulationOptions lcm_options = options;
274  if (lcm_options.realtime_factor < 0.0) lcm_options.realtime_factor = 1.0;
275  simulate(*lcm_sys, t0, tf, x0, lcm_options);
276 
277  if (has_lcm_input) {
278  // shutdown the lcm thread
279  lcm_loop.stop = true;
