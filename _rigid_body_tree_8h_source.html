733  * the Jacobian on the joints that are on the kinematic path; if we want to
734  * reconstruct the full Jacobian on all joints, then we should call this
735  * method.
736  */
737  int ncols = in_terms_of_qdot ? num_positions_ : num_velocities_;
738  Eigen::Matrix<typename Derived::Scalar, Derived::RowsAtCompileTime,
739  Eigen::Dynamic> full(compact.rows(), ncols);
740  full.setZero();
741  int compact_col_start = 0;
742  for (std::vector<int>::const_iterator it = joint_path.begin();
743  it != joint_path.end(); ++it) {
744  RigidBody& body = *bodies[*it];
745  int ncols_joint = in_terms_of_qdot ? body.getJoint().getNumPositions()
746  : body.getJoint().getNumVelocities();
747  int col_start =
748  in_terms_of_qdot ? body.position_num_start : body.velocity_num_start;
749  full.middleCols(col_start, ncols_joint) =
750  compact.middleCols(compact_col_start, ncols_joint);
751  compact_col_start += ncols_joint;
752  }
753  return full;
754  }
