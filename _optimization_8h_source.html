609  std::shared_ptr<Constraint> AddPolynomialConstraint(
610  const VectorXPoly& polynomials,
611  const std::vector<Polynomiald::VarType>& poly_vars,
612  const Eigen::VectorXd& lb, const Eigen::VectorXd& ub,
613  const VariableList& vars) {
614  // Polynomials that are actually affine (a sum of linear terms + a
615  // constant) can be special-cased. Other polynomials are treated as
616  // generic for now.
617  // TODO(ggould-tri) There may be other such special easy cases.
618  bool all_affine = true;
619  for (int i = 0; i < polynomials.rows(); i++) {
620  if (!polynomials[i].isAffine()) {
621  all_affine = false;
622  break;
623  }
624  }
625  if (all_affine) {
626  Eigen::MatrixXd linear_constraint_matrix =
627  Eigen::MatrixXd::Zero(polynomials.rows(), poly_vars.size());
628  Eigen::VectorXd linear_constraint_lb = lb;
629  Eigen::VectorXd linear_constraint_ub = ub;
630  for (int poly_num = 0; poly_num < polynomials.rows(); poly_num++) {
631  for (const auto& monomial : polynomials[poly_num].getMonomials()) {
632  if (monomial.terms.size() == 0) {
633  linear_constraint_lb[poly_num] -= monomial.coefficient;
634  linear_constraint_ub[poly_num] -= monomial.coefficient;
635  } else if (monomial.terms.size() == 1) {
636  const Polynomiald::VarType term_var = monomial.terms[0].var;
637  int var_num = (std::find(poly_vars.begin(), poly_vars.end(),
638  term_var) -
639  poly_vars.begin());
640  DRAKE_ASSERT(var_num < static_cast<int>(poly_vars.size()));
641  linear_constraint_matrix(poly_num, var_num) = monomial.coefficient;
642  } else {
643  DRAKE_ABORT(); // Can't happen (unless isAffine() lied to us).
644  }
645  }
646  }
647  if (ub == lb) {
648  std::shared_ptr<LinearEqualityConstraint> constraint(
649  new LinearEqualityConstraint(linear_constraint_matrix,
650  linear_constraint_ub));
651  AddLinearEqualityConstraint(constraint, vars);
652  return constraint;
653  } else {
654  std::shared_ptr<LinearConstraint> constraint(
655  new LinearConstraint(linear_constraint_matrix,
656  linear_constraint_lb,
657  linear_constraint_ub));
658  AddLinearConstraint(constraint, vars);
659  return constraint;
660  }
661  } else {
662  std::shared_ptr<PolynomialConstraint> constraint(
663  new PolynomialConstraint(polynomials, poly_vars, lb, ub));
664  AddGenericConstraint(constraint, vars);
665  return constraint;
666  }
667  }
