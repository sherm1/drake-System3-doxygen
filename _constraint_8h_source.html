227  /* updateConstraint
228  * @brief change the parameters of the constraint (A and b), but not the
229  *variable associations
230  *
231  * note that A and b can change size in the rows only (representing a
232  *different number of linear constraints, but on the same decision variables)
233  */
234  template <typename DerivedA, typename DerivedB>
235  void updateConstraint(const Eigen::MatrixBase<DerivedA>& Aeq,
236  const Eigen::MatrixBase<DerivedB>& beq) {
237  DRAKE_ASSERT(Aeq.rows() == beq.rows());
238  if (Aeq.cols() != A_.cols())
239  throw std::runtime_error("Can't change the number of decision variables");
240  A_.resize(Aeq.rows(), Eigen::NoChange);
241  A_ = Aeq;
242  lower_bound_.conservativeResize(beq.rows());
243  lower_bound_ = beq;
244  upper_bound_.conservativeResize(beq.rows());
245  upper_bound_ = beq;
246  }
